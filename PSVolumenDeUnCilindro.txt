ALGORITMO Volumen de un cilindro											
// Este algoritmo calcula el volument de un cilindro
MODULOS INCLUIDOS											
				 ENTRADA-SALIDA
CONSTANTES 
		  FLOAT: pi
VARIABLES
		 FLOAT: radio, altura, volumen				
INICIO											
	  MOSTRAR("Ingrese los siguientes datos (metros): ")	
	  MOSTRAR("Radio: ")	
	  LEER(radio)	
	  MOSTRAR("Altura: ")	  
	  LEER(altura)
	  
	  volumen <- pi*radio*radio*altura	
	  
	  MOSTRAR("El volumen del cilindro es: ", volumen)								
FIN ALGORITMO Convertir
