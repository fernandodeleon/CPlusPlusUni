ALGORITMO Convertir											
// Este algoritmo convierte la temperatura de celcius a fahrenheit
MODULOS INCLUIDOS											
				 ENTRADA-SALIDA
CONSTANTES //No hay											
VARIABLES											
		 ENTERO: grados_celcius
		 FLOAT: grados_fahrenheit						
INICIO											
	  MOSTRA("Ingresa grados centigrados: ", + CDL)								
	  LEER(grados_celcius)								
	  grados_fahrenheit <- (grados_celcius*9/5)+(32)						
	  MOSTRAR(grados_celcius, "C equivale a", grados_fahrenheit, "F")								
FIN ALGORITMO Convertir
